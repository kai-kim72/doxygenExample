208void MicomIf::processPe2Message(uint8_t catNo, uint8_t *msg, uint16_t msgLen) {

210 uint8_t state = msg[VEHICLE_PROTOCOL_STATE_IDX] & 0x7F;
211 uint8_t cmdId = msg[VEHICLE_PROTOCOL_CMD_ID_IDX];
212
213 if(state == VEHICLE_PROTOCOL_STATE_MCU_TO_CPU_ACK) {
214 IpcTxStatus &status = ipcTxStatus[(cmdId & 0x7F)];

222 //LOG_D("[%s:%d] debug..", __PRETTY_FUNCTION__, __LINE__);

224 ipcTxAck(cmdId);
225 }
226 switch(cmdId) {
227 case VEHICLE_PROTOCOL_CMD_ID_IGN2:

270 break;

417int MicomIf::ipcTxData(uint8_t *data, uint16_t dataSize, bool async) {
418 int result = -1;
419 qDebug("[%s:%d] dataSize = %d, async = %d", __PRETTY_FUNCTION__, __LINE__, dataSize, async);
420
421 if((data != nullptr) && (dataSize >= VEHICLE_PROTOCOL_HEADER_SIZE)) {
422 IpcTxStatus &status = ipcTxStatus[(data[VEHICLE_PROTOCOL_CMD_ID_IDX] & 0x7F)];
423 status.txLock.lock();
424 status.dataLock.lock();
425 while(status.f.valid() == true) {
426 qDebug("[%s:%d] future is valid!!!", __PRETTY_FUNCTION__, __LINE__);
427 status.f.get();
428 }
429
430#ifdef USE_MICOM_IF
431 tc_ipc_ctrl_msg_send(ipcHandle, TCC_IPC_CMD_IK_PE2_MESSAGE, TCC_IPC_CMD_IK_PE2_MESSAGE_CAT0, data, dataSize);
432#endif
433 MicomIf::printBuffer("[IPC_TX]", TCC_IPC_CMD_IK_PE2_MESSAGE, TCC_IPC_CMD_IK_PE2_MESSAGE_CAT0, data, dataSize);
434 status.waitAck = true;
435 if(status.data == nullptr) {
436 status.data = new uint8_t[dataSize];
437 status.dataSize = dataSize;
438 }
439 std::memcpy(status.data, data, static_cast<std::size_t>(dataSize));
440 status.dataLock.unlock();
441 if(async == true) {
442 status.f = std::async(std::launch::async, &MicomIf::ipcTxDataRetry, &status);
443 }
444 else {
445 ipcTxDataRetry(&status);
446 }
447 result = 0;
448 }
449
450 return result;

479void MicomIf::ipcTxDataRetry(IpcTxStatus *status) {
480 qDebug("[%s:%d]", __PRETTY_FUNCTION__, __LINE__);
481 int retryCount = 0;
482
483 if(status != nullptr) {
484 std::unique_lock<std::mutex> lck(status->dataLock);
485 do {
486 status->ackCv.wait_for(lck, std::chrono::milliseconds(IPC_TX_ACK_TIMEOUT_MS));
487 if(status->waitAck == true) {
488 if((status->data != nullptr) && (status->dataSize >= VEHICLE_PROTOCOL_HEADER_SIZE)) {
489 if(retryCount++ < IPC_TX_RETRY_MAX) {
490#ifdef USE_MICOM_IF
491 tc_ipc_ctrl_msg_send(ipcHandle, TCC_IPC_CMD_IK_PE2_MESSAGE, TCC_IPC_CMD_IK_PE2_MESSAGE_CAT0, status->data, status->dataSize);

494 }
495 else {
496 qDebug("[%s:%d] no ack for 3 retry!!!.. cmdId=%d", __PRETTY_FUNCTION__, __LINE__, status->data[VEHICLE_PROTOCOL_CMD_ID_IDX]);
497 status->waitAck = false;
498 }
499 }
500 else {
501 qDebug("[%s:%d] data is null or less data!!!.. cmdId=%d", __PRETTY_FUNCTION__, __LINE__, status->data[VEHICLE_PROTOCOL_CMD_ID_IDX]);
